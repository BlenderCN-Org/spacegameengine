MACRO (ADD_SGE_TEST name)
	ADD_EXECUTABLE (sge${name}test ${name}.cpp)
	TARGET_LINK_LIBRARIES (sge${name}test sgecore)
	INSTALL (TARGETS sge${name}test DESTINATION ${CMAKE_INSTALL_PREFIX}/bin)
ENDMACRO ()
