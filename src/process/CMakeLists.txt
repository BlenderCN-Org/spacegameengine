MACRO(
	ADD_SGE_PROCESS_INCLUDE_SOURCE_GROUP
	SOURCEPATH
)
	ADD_SGE_SOURCE_GROUP(
		"include\\\\\\\\sge\\\\\\\\process" #cmake bitch
		${CMAKE_SOURCE_DIR}/include/sge/process
		"${SOURCEPATH}"
		*.hpp
		TEMPFILES
	)

	SET(SGEPROCESSFILES ${SGEPROCESSFILES} ${TEMPFILES})
ENDMACRO()

ADD_SGE_PROCESS_INCLUDE_SOURCE_GROUP("")

	ADD_SGE_PROCESS_SOURCE_GROUP
	SOURCEPATH
)
	ADD_SGE_SOURCE_GROUP(
		"src\\\\\\\\process"
		""
		"${SOURCEPATH}"
		*.?pp
		TEMPFILES	
	)

	SET(SGEPROCESSFILES ${SGEPROCESSFILES} ${TEMPFILES})
ENDMACRO()
