MACRO(
	ADD_SGE_ICONV_INCLUDE_SOURCE_GROUP
	SOURCEPATH
)
	ADD_SGE_SOURCE_GROUP(
		"include\\\\\\\\sge\\\\\\\\iconv" #cmake bitch
		${CMAKE_SOURCE_DIR}/include/sge/iconv
		"${SOURCEPATH}"
		*.hpp
		TEMPFILES
	)

	SET(SGEICONVFILES ${SGEICONVFILES} ${TEMPFILES})
ENDMACRO()

ADD_SGE_ICONV_INCLUDE_SOURCE_GROUP("")

MACRO(
	ADD_SGE_ICONV_SOURCE_GROUP
	SOURCEPATH
)
	ADD_SGE_SOURCE_GROUP(
		"src\\\\\\\\iconv"
		""
		"${SOURCEPATH}"
		*.?pp
		TEMPFILES	
	)

	SET(SGEICONVFILES ${SGEICONVFILES} ${TEMPFILES})
ENDMACRO()

ADD_SGE_ICONV_SOURCE_GROUP("")
