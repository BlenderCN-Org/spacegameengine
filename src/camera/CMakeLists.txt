MACRO(
	ADD_SGE_CAMERA_INCLUDE_SOURCE_GROUP
	SOURCEPATH
)
	ADD_SGE_SOURCE_GROUP(
		"include\\\\\\\\sge\\\\\\\\camera" #cmake bitch
		${CMAKE_SOURCE_DIR}/include/sge/camera
		"${SOURCEPATH}"
		*.hpp
		TEMPFILES
	)

	SET(SGECAMERAFILES ${SGECAMERAFILES} ${TEMPFILES})
ENDMACRO()

ADD_SGE_CAMERA_INCLUDE_SOURCE_GROUP("")
ADD_SGE_CAMERA_INCLUDE_SOURCE_GROUP(projection)
ADD_SGE_CAMERA_INCLUDE_SOURCE_GROUP(gizmo)

)
