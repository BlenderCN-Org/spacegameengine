MACRO(
	ADD_SGE_DEVIL_SOURCE_GROUP
	SOURCEPATH
)
	ADD_SGE_SOURCE_GROUP(
		"src\\\\\\\\plugins\\\\\\\\devil"
		""
		"${SOURCEPATH}"
		*.?pp
		TEMPFILES
	)

	SET(SGEDEVILFILES ${SGEDEVILFILES} ${TEMPFILES})
ENDMACRO()

ADD_SGE_DEVIL_SOURCE_GROUP("")
ADD_SGE_DEVIL_SOURCE_GROUP(src)

INCLUDE_DIRECTORIES(${IL_INCLUDE_DIR})

ADD_LIBRARY (sgedevil SHARED ${SGEDEVILFILES})

TARGET_LINK_LIBRARIES (
	sgedevil
	sgecore
	${IL_LIBRARIES}
	${ILU_LIBRARIES}
)

INSTALL (TARGETS sgedevil DESTINATION lib/sge)
