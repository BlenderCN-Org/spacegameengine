MACRO(
	ADD_SGE_PLUGIN
	PLUGIN_NAME
	ADDITONAL_DEPS
)
	SET(
		SGE_PLUGIN_NAME
		sge${PLUGIN_NAME}
	)

	STRING(
		TOUPPER
		${PLUGIN_NAME}
		UPPER_PLUGIN_NAME
	)

	FCPPT_UTILS_APPEND_SOURCE_DIR_AND_MAKE_GROUPS(
		"${SGE_${UPPER_PLUGIN_NAME}_FILES}"
		SGE_${UPPER_PLUGIN_NAME}_FILES_ABS
	)

	ADD_LIBRARY(
		${SGE_PLUGIN_NAME}
		SHARED
		${SGE_${UPPER_PLUGIN_NAME}_FILES_ABS}
	)

	CHECK_LIBRARY_DEPS(
		"${PLUGIN_NAME}"
		"${ADDITONAL_DEPS}"
	)

	TARGET_LINK_LIBRARIES(
		${SGE_PLUGIN_NAME}
		sgecore
		${ADDITONAL_DEPS}
	)

	INSTALL(
		TARGETS
		${SGE_PLUGIN_NAME}
		DESTINATION
		${PLUGIN_INSTALL_DIR}
	)
ENDMACRO()
