MACRO(
	ADD_SGE_PNG_SOURCE_GROUP
	SOURCEPATH
)
	ADD_SGE_SOURCE_GROUP(
		"src\\\\\\\\plugins\\\\\\\\png"
		""
		"${SOURCEPATH}"
		*.?pp
		TEMPFILES
	)

	SET(SGEPNGFILES ${SGEPNGFILES} ${TEMPFILES})
ENDMACRO()

ADD_SGE_PNG_SOURCE_GROUP("")
ADD_SGE_PNG_SOURCE_GROUP(src)

INCLUDE_DIRECTORIES(${PNG_INCLUDE_DIR})

ADD_DEFINITIONS(${PNG_DEFINITIONS})

ADD_LIBRARY (sgelibpng SHARED ${SGEPNGFILES})

TARGET_LINK_LIBRARIES (sgelibpng sgecore ${PNG_LIBRARY})

INSTALL (TARGETS sgelibpng DESTINATION lib/sge)
