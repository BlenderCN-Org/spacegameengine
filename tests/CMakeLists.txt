function(
	add_sge_test
	PATHNAME
)
	set(
		OPTION_ARGS
		NO_CODE
	)

	set(
		MULTI_ARGS
		SGE_DEPS
		ADDITIONAL_DEPS
		INCLUDE_DIRS
	)

	cmake_parse_arguments(
		""
		"${OPTION_ARGS}"
		""
		"${MULTI_ARGS}"
		${ARGN}
	)

	sge_check_optional_deps(
		"${PATHNAME}"
		"${_SGE_DEPS}"
	)

	transform_sge_link_targets(
		"${_SGE_DEPS}"
		SGE_LIBRARIES_RESULT
	)

	if(
		_NO_CODE
	)
		set(
			NO_CODE
			"NO_CODE"
		)
	endif()

	fcppt_utils_add_test(
		"${PATHNAME}"
		${NO_CODE}
		LINK_LIBS
			${SGE_LIBRARIES_RESULT}
			${_ADDITIONAL_DEPS}
		INCLUDE_DIRS
			${_INCLUDE_DIRS}
	)
endfunction()
