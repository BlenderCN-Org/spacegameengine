MACRO (ADD_SGE_EXAMPLE name)
	ADD_EXECUTABLE (sge${name}example ${name}.cpp)
	TARGET_LINK_LIBRARIES (sge${name}example sgecore)
	INSTALL (TARGETS sge${name}example DESTINATION ${CMAKE_INSTALL_PREFIX}/bin)
ENDMACRO ()
